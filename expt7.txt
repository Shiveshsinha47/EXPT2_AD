package com.helloworld.sleeptracker

import android.app.TimePickerDialog import android.content.SharedPreferences import android.os.Bundle
import android.os.SystemClock import android.widget.*
import androidx.appcompat.app.AlertDialog
import androidx.appcompat.app.AppCompatActivity class MainActivity : AppCompatActivity() {
lateinit var chronometer: Chronometer lateinit var stageText: TextView lateinit var avgSleep: TextView

lateinit var startBtn: Button lateinit var stopBtn: Button lateinit var resetBtn: Button lateinit var settingsBtn: Button
 
lateinit var prefs: SharedPreferences var running = false
override fun onCreate(savedInstanceState: Bundle?) {

super.onCreate(savedInstanceState) setContentView(R.layout.activity_main)

chronometer = findViewById(R.id.chronometer) stageText = findViewById(R.id.sleepStage) avgSleep = findViewById(R.id.avgSleep)

startBtn = findViewById(R.id.startBtn) stopBtn = findViewById(R.id.stopBtn) resetBtn = findViewById(R.id.resetBtn) settingsBtn = findViewById(R.id.settingsBtn)
prefs = getSharedPreferences("sleepData", MODE_PRIVATE) avgSleep.text = getAverageSleep()
stopBtn.isEnabled = false

startBtn.setOnClickListener { startSleep() } stopBtn.setOnClickListener { stopSleep() } resetBtn.setOnClickListener { resetSleep() } settingsBtn.setOnClickListener { openGoalPicker() }

chronometer.setOnChronometerTickListener {
val elapsed = (SystemClock.elapsedRealtime() - chronometer.base) / 1000 when {
elapsed >= 5 -> stageText.text = "Sleep Stage: Deep Sleep" elapsed >= 2 -> stageText.text = "Sleep Stage: Light Sleep" else -> stageText.text = "Sleep Stage: Falling Asleep"
}
}
}

fun startSleep() { if (!running) {
chronometer.base = SystemClock.elapsedRealtime() chronometer.start()

running = true startBtn.isEnabled = false stopBtn.isEnabled = true
}
}
 
fun stopSleep() { if (running) {
chronometer.stop()

val duration = (SystemClock.elapsedRealtime() - chronometer.base) / 1000 saveSleep(duration)
avgSleep.text = getAverageSleep()

running = false startBtn.isEnabled = true stopBtn.isEnabled = false
showSummary(duration)
}
}

fun resetSleep() {

chronometer.base = SystemClock.elapsedRealtime() stageText.text = "Sleep Stage: Awake"

Toast.makeText(this, "Session Reset", Toast.LENGTH_SHORT).show()
}

fun saveSleep(duration: Long) {

val prev = prefs.getLong("totalSleep", 0) val count = prefs.getInt("days", 0)

prefs.edit()
.putLong("totalSleep", prev + duration)
.putInt("days", count + 1)
.apply()
}

fun getAverageSleep(): String {

val total = prefs.getLong("totalSleep", 0) val days = prefs.getInt("days", 1)
val avg = total / days val hours = avg / 3600
val minutes = (avg % 3600) / 60

return "Average Sleep: ${hours}h ${minutes}m"
}

fun openGoalPicker() {
 
val picker = TimePickerDialog( this,
{ _, hour, minute ->

Toast.makeText( this,
"Bedtime Goal Set: $hour:$minute", Toast.LENGTH_LONG
).show()
},
22,
0,
true
)

picker.show()
}
fun showSummary(duration: Long) { val hours = duration / 3600
val minutes = (duration % 3600) / 60 val seconds = duration % 60

val sleepType = when {
duration >= 5 -> "Deep Sleep" duration >= 2 -> "Light Sleep" else -> "Falling Asleep"
}

val message = """
Sleep Duration: ${hours}h ${minutes}m ${seconds}s

Sleep Stage Reached:
$sleepType
Great job tracking your sleep! """.trimIndent()

AlertDialog.Builder(this)
.setTitle("Sleep Summary")
.setMessage(message)
.setPositiveButton("OK") { dialog, _ ->
dialog.dismiss()
}
.show()
}
}
